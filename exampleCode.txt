int sort(int arg){
	//mao pao pai xu
	int a[10];
	int i=0;
	int j=0;
	int temp;
	while(i<10){
		while(j<10-i-1){
			if(a[j]>a[j+1]){
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
			j=j+1;
		}
		j=0;
		i=i+1;
	}
	return 0;
}

int fact(int base){
	if(base == 0){
		return 1;
	}
	else{
		return base*fact(base-1);
	}
}
